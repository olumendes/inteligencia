### Estrutura de Licitações (mockData)
```typescript
interface Licitacao {
  id: string;
  titulo: string;
  valor: number;
  status: "aberta" | "encerrada" | "resultado";
  modalidade: string;
  orgao: string;
  dataAbertura: string;
  dataEncerramento: string;
  descricao: string;
  salva: boolean;
}
```

### Estrutura de Alertas
```typescript
interface Alerta {
  id: string;
  palavra_chave: string;
  ativo: boolean;
  frequencia: "diaria" | "semanal" | "mensal";
  ultimaVerificacao: string;
}
```
